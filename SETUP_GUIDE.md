#!/bin/bash
# =====================================================================
# FLUTTER PROJECT SETUP & DEPLOYMENT GUIDE
# Production-Ready for Web, Android, and iOS
# =====================================================================

# ============ PREREQUISITES ============
# Ensure you have:
# - Flutter SDK 3.10.1 or higher
# - Android SDK (for Android builds)
# - Xcode 14+  (for iOS builds)
# - Chrome or a web browser (for web testing)
# - Firebase Console Project created




# ============ 1. INITIAL SETUP ============
# Run this once after cloning the project

echo "Step 1: Installing dependencies..."
cd D:\matjark
flutter clean
flutter pub get

echo "Step 2: Getting Firebase credentials..."
# This requires Firebase CLI installed (https://firebase.google.com/docs/cli)
# Run this command to generate firebase_options.dart with real credentials:
#   flutterfire configure




# ============ 2. RUNNING ON WEB ============
echo "Step 3: Testing on Web (Chrome)..."
flutter run -d chrome

# ⚠️ NOTE:
# - Web requires real Firebase credentials in firebase_options.dart
# - Update lib/firebase_options.dart with your web credentials after running flutterfire configure
# - Phone auth on web uses reCAPTCHA (Firebase handles this automatically)




# ============ 3. RUNNING ON ANDROID ============
echo "Step 4: Building for Android..."

# First, ensure Android setup is correct:
flutter doctor -v

# Run on connected Android device:
flutter run -d android

# Or build APK for testing:
flutter build apk --release

# APK location: build/app/outputs/apk/release/app-release.apk




# ============ 4. RUNNING ON iOS ============
echo "Step 5: Building for iOS..."

# First, ensure iOS setup is correct:
flutter doctor -v

# CocoaPods setup (run once):
cd ios
pod repo update
pod install
cd ..

# Run on iOS simulator or device:
flutter run -d ios

# Or build IPA for App Store:
flutter build ios --release




# ============ 5. FIXING COMMON ISSUES ============

# A) macOS facebook_auth_desktop error
#    Already fixed: flutter_facebook_auth removed from pubspec.yaml
#    Reason: Web-only projects don't need Facebook auth for desktop
#    If you need Facebook auth, use: flutter_facebook_auth: ^6.0.0

# B) Firebase initialization failed
#    Run: flutterfire configure
#    This will:
#    - Prompt you to select Firebase project
#    - Generate lib/firebase_options.dart with correct credentials
#    - Update android/build.gradle and iOS files

# C) KeyboardType not recognized
#    Fixed: keyboardType is a parameter of TextFormField, NOT InputDecoration
#    Example:
#      TextFormField(
#        keyboardType: TextInputType.email,  // ✓ Correct
#        decoration: InputDecoration(labelText: 'Email'),
#      )

# D) CardTheme deprecated error
#    Fixed: Use CardThemeData (not CardTheme) in Flutter 3.10+
#    The theme.dart now uses CardThemeData for Material Design 3 compliance

# E) EasyLocalization missing keys
#    Add missing keys to:
#    - assets/translations/en.json
#    - assets/transitions/ar.json
#    Check login_screen.dart for all used keys




# ============ 6. PROJECT STRUCTURE OVERVIEW ============
# 
# lib/
# ├── main.dart                 # Application entry point with Firebase init
# ├── firebase_options.dart     # Generated by: flutterfire configure
# ├── core/
# │   ├── theme.dart           # Material Design 3 themes (Light/Dark)
# │   ├── constants.dart        # App constants (titles, strings)
# │   ├── auth_result.dart      # Result type system for error handling
# │   └── firebase_status.dart  # Firebase availability flag
# ├── models/
# │   └── user_model.dart       # AppUser and UserRole enums
# ├── services/
# │   └── auth_service.dart     # Firebase Auth wrapper (email, phone, Google)
# ├── providers/
# │   ├── auth_provider.dart    # Auth state management (ChangeNotifier)
# │   └── role_provider.dart    # Role-based navigation (ChangeNotifierProxyProvider)
# └── screens/
#     ├── splash_screen.dart
#     ├── auth/
#     │   ├── login_screen.dart       # Email/Phone toggle with OTP
#     │   └── register_screen.dart    # Registration form
#     ├── customer/
#     ├── seller/
#     ├── supplier/
#     └── admin/




# ============ 7. BUILDING FOR PRODUCTION ============

echo "Step 6: Building for production..."

# Web Production Build:
flutter build web --release
# Output: build/web/

# Android Production Build (APK):
flutter build apk --release
# Output: build/app/outputs/apk/release/app-release.apk

# Android Production Build (AAB for Play Store):
flutter build appbundle --release
# Output: build/app/outputs/bundle/release/app-release.aab

# iOS Production Build:
flutter build ipa --release
# Output: build/ios/ipa/




# ============ 8. PRODUCTION CHECKLIST ============
# 
# [ ] firebase_options.dart has real Firebase credentials
# [ ] Build variant/flavor is set to production
# [ ] All sensitive data is in Firebase/Firestore (not hardcoded)
# [ ] Phone auth reCAPTCHA domain is configured in Firebase console for web
# [ ] App icons and splash screens are updated
# [ ] localization keys are complete in both en.json and ar.json
# [ ] Error logging is enabled (debugPrint for debug, proper logging for production)
# [ ] Rate limiting is configured for auth endpoints
# [ ] CORS is properly configured if API calls are needed
# [ ] SSL certificate pinning is considered for production
# [ ] Review Firebase security rules for Firestore and Storage




# ============ 9. DEPLOYMENT ============

# Google Play Store (Android):
# - Build appbundle: flutter build appbundle --release
# - Upload to: https://play.google.com/console
# - Set up signing from Play Console (app signing)

# Apple App Store (iOS):
# - Build ipa: flutter build ipa --release
# - Use Xcode Organizer or Transporter to upload
# - Configure developer.apple.com provisioning profiles

# Firebase Hosting (Web):
# - Install Firebase CLI: npm install -g firebase-tools
# - flutter build web --release
# - firebase deploy




# ============ 10. MONITORING & MAINTENANCE ============

# Monitor crashes:
# - Firebase Console > Crashlytics tab

# Monitor authentication:
# - Firebase Console > Authentication > All users tab

# Monitor database:
# - Firebase Console > Firestore Database

# Update Flutter packages:
flutter pub upgrade

# Check for security vulnerabilities:
flutter pub audit




echo "Setup complete! Your production-ready Flutter app is ready to test and deploy."
